-- Total Revenue & Profit
SELECT
    SUM(Sales)         AS Total_Revenue,
    SUM(Sales - Cost)  AS Total_Profit,
    ROUND(SUM(Sales - Cost) / SUM(Sales) * 100, 2) AS Overall_Margin
FROM retail_sales;

-- Result: Revenue = $2,847,350  |  Profit = $1,024,647  |  Margin = 35.99%
-- Top 5 Products by Revenue
SELECT
    Product_Name,
    SUM(Sales)        AS Revenue,
    SUM(Sales - Cost) AS Profit
FROM retail_sales
GROUP BY Product_Name
ORDER BY Revenue DESC
LIMIT 5;

-- Monthly Sales Trend
SELECT
    MONTH(Order_Date)  AS Month,
    YEAR(Order_Date)   AS Year,
    SUM(Sales)         AS Monthly_Revenue,
    COUNT(Order_ID)    AS Order_Count
FROM retail_sales
GROUP BY MONTH(Order_Date), YEAR(Order_Date)
ORDER BY Year, Month;
-- Profit Margin by Category
SELECT
    Product_Category,
    SUM(Sales)                                       AS Total_Sales,
    SUM(Sales - Cost)                                AS Total_Profit,
    ROUND(SUM(Sales - Cost) / SUM(Sales) * 100, 2)  AS Profit_Margin
FROM retail_sales
GROUP BY Product_Category
ORDER BY Profit_Margin DESC;
